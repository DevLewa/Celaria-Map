Datatypes

Type Description
int_u8 unsigned byte (8 bit integer)
int_u16 unsigned short (16 bit integer)
int_u32 unsigned int (32 bit integer)
float_f32 32 bit floating point value (float)
float_f64 64 bit floating point value (double)
string(x) string consisting of x elements of unsigned bytes

string(x)

Pseudocode 
		
String s = ""
for(i = 0;i < x;i++){
	s+=toChar(int_u8) //toChar converts byte to character according to ascii conversion

}
return s

Cmap format (.cmap)

		
fileID = string(11) //file identifier. should equal to "celaria_map"
version = int_u8

mapNameLength = int_u8
map_name = string(mapNameLength)

int_u8 //unused byte

if(version == 0){
	int_u8 //unused byte
}
numCheckpoints = int_u8 //number of checkpoints

//platin times
repeat(numCheckpoints){
	int_u32 //checkpoint time
}
//gold times
repeat(numCheckpoints){
	int_u32 //checkpoint time
}
//silver times
repeat(numCheckpoints){
	int_u32 //checkpoint time
}
//bronze times
repeat(numCheckpoints){
	int_u32 //checkpoint time
}

sunRotation_horizontal = float_f32
sunRotation_vertical = float_f32

//preview camera
previewCam_xfrom = float_f64
previewCam_yfrom = float_f64
previewCam_zfrom = float_f64

previewCam_xto = float_f64
previewCam_yto = float_f64
previewCam_zto = float_f64

numEntities = int_u32

repeat(numEntities){
	entityType = int_u8
	switch(entityType){
		case 0: //block
			blockType = int_u8
			if(version == 0){
				int_u8 //unused
			}
			if(version <= 1){
				xPos = int_s32
				yPos = int_s32
				zPos = int_u32

				xScale = int_u32
				yScale = int_u32
				zScale = int_u32
			}else{
				xPos = float_f64
				yPos = float_f64
				zPos = float_f64

				xScale = float_f64
				yScale = float_f64
				zScale = float_f64					
			}
			zRotation = float_f32
			if(blockType == 5){
				checkpoint_id = int_u8
			}
		break;

		case 1://sphere
			if(version <= 1){
				xPos = int_s32
				yPos = int_s32
				if(version == 0){
					zPos = int_s32
				}else{
					zPos = int_u32
				}
			}else{
				xPos = float_f64
				yPos = float_f64
				zPos = float_f64
			}

		break;
		case 2://player start
			int_u8 //unused
			if(version <= 1){
				xPos = int_s32
				yPos = int_s32
				if(version == 0){
					zPos = int_s32
				}else{
					zPos = int_u32
				}
			}else{
				xPos = float_f64
				yPos = float_f64
				zPos = float_f64
			}
			zRotation = float_f32
		break;

		case 3: //barrier wall
			//only valid if version is >= 2
			int_u8 //unused (should be 0)
			xPos = float_f64
			yPos = float_f64
			zPos = float_f64
			xScale = float_f64
			zScale = float_f64
			zRotation = float_f32
		break;

		case 4: //barrier floor
			//only valid if version is >= 2
			int_u8 //unused (should be 0)
			xPos = float_f64
			yPos = float_f64
			zPos = float_f64
			xScale = float_f64
			yScale = float_f64
			zRotation = float_f32
		break;
		case 128: //empty object
			empty_id = int_u8
			if (version <= 1){
				xPos = int_s32
				yPos = int_s32
				zPos = int_u32

				xScale = int_u32
				yScale = int_u32
				zScale = int_u32

			}else{
				xPos = float_f64
				yPos = float_f64
				zPos = float_f64

				xScale = float_f64
				yScale = float_f64
				zScale = float_f64					
			}

			zRotation = float_f32
		break;


	}


}
